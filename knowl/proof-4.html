\begin{align*} x \in \overline{A \cup B} & \Longleftrightarrow x \notin A \cup B \Longleftrightarrow x \notin A \text{かつ} x \notin B \Longleftrightarrow x \in \overline{A} \text{かつ} x \in \overline{B} \Longleftrightarrow x \in \overline{A} \cap \overline{B}\\ x \in \overline{A \cap B} & \Longleftrightarrow x \notin A \cap B \Longleftrightarrow x \notin A \text{または} x \notin B \Longleftrightarrow x \in \overline{A} \text{または} x \in \overline{B} \Longleftrightarrow x \in \overline{A} \cup \overline{B}\\ \overline{A \cup B \cup C} &= \overline{A \cup (B \cup C)} = \overline{A} \cap \overline{B \cup C} = \overline{A} \cap (\overline{B} \cap \overline{C}) = \overline{A} \cap \overline{B} \cap \overline{C} \end{align*}
